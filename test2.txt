# The game board is represented by a dictionary
theBoard = {
    '7': ' ', '8': ' ', '9': ' ',
    '4': ' ', '5': ' ', '6': ' ',
    '1': ' ', '2': ' ', '3': ' '
}

def printBoard(board):
    """Prints the current state of the Tic-Tac-Toe board."""
    print(board['7'] + ' | ' + board['8'] + ' | ' + board['9'])
    print('-+-+-')
    print(board['4'] + ' | ' + board['5'] + ' | ' + board['6'])
    print('-+-+-')
    print(board['1'] + ' | ' + board['2'] + ' | ' + board['3'])
    print('\n')

def game():
    """Main game loop for Tic-Tac-Toe."""
    turn = 'X'
    count = 0
    
    while count < 9:
        printBoard(theBoard)
        print(f"It's your turn, {turn}. Move to which place? (1-9)")
        
        move = input()

        # Input validation: check if the spot is already taken
        if theBoard[move] == ' ':
            theBoard[move] = turn
            count += 1
        else:
            print("That place is already filled.\nMove to which place?")
            continue # Skip the rest of the loop and ask for input again

        # Check for a win after 5 or more moves have been made
        if count >= 5:
            # Check for horizontal wins
            if theBoard['7'] == theBoard['8'] == theBoard['9'] != ' ':
                printBoard(theBoard)
                print(f"\n**** {turn} won! ****")
                break
            elif theBoard['4'] == theBoard['5'] == theBoard['6'] != ' ':
                printBoard(theBoard)
                print(f"\n**** {turn} won! ****")
                break
            elif theBoard['1'] == theBoard['2'] == theBoard['3'] != ' ':
                printBoard(theBoard)
                print(f"\n**** {turn} won! ****")
                break
            # Check for vertical wins
            elif theBoard['7'] == theBoard['4'] == theBoard['1'] != ' ':
                printBoard(theBoard)
                print(f"\n**** {turn} won! ****")
                break
            elif theBoard['8'] == theBoard['5'] == theBoard['2'] != ' ':
                printBoard(theBoard)
                print(f"\n**** {turn} won! ****")
                break
            elif theBoard['9'] == theBoard['6'] == theBoard['3'] != ' ':
                printBoard(theBoard)
                print(f"\n**** {turn} won! ****")
                break
            # Check for diagonal wins
            elif theBoard['7'] == theBoard['5'] == theBoard['3'] != ' ':
                printBoard(theBoard)
                print(f"\n**** {turn} won! ****")
                break
            elif theBoard['9'] == theBoard['5'] == theBoard['1'] != ' ':
                printBoard(theBoard)
                print(f"\n**** {turn} won! ****")
                break

        # If no winner yet, switch players
        turn = 'O' if turn == 'X' else 'X'

    # If the loop finishes without a win, it's a tie
    if count == 9 and not (theBoard['7'] == theBoard['8'] == theBoard['9'] != ' ' or theBoard['4'] == theBoard['5'] == theBoard['6'] != ' ' or theBoard['1'] == theBoard['2'] == theBoard['3'] != ' ' or theBoard['7'] == theBoard['4'] == theBoard['1'] != ' ' or theBoard['8'] == theBoard['5'] == theBoard['2'] != ' ' or theBoard['9'] == theBoard['6'] == theBoard['3'] != ' ' or theBoard['7'] == theBoard['5'] == theBoard['3'] != ' ' or theBoard['9'] == theBoard['5'] == theBoard['1'] != ' '):
        printBoard(theBoard)
        print("\nGame Over.")
        print("It's a Tie!!")

if __name__ == "__main__":
    game()
